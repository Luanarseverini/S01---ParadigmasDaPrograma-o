#Exercicio 2 - Hierarquia de carros

class Carro
  def initialize(modelo)
    @modelo = modelo
  end

  def descrever
    "Carro, modelo: #{@modelo}"
  end
end

class CarroEsportivo < Carro
  def descrever
    "Carro esportivo, modelo: #{@modelo}"
  end
end

class CarroSedan < Carro
  def descrever
    "Carro sedan, modelo: #{@modelo}"
  end
end
